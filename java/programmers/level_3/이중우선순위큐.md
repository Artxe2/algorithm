class Solution {
    public int[] solution(String[] operations) {
        int size = 0;
        int[] queue = new int[operations.length];

        for (String s : operations) {
            if (s.charAt(0) == 'I') {
                add(queue, size++, parseInt(s));
            } else if (s.charAt(2) == '-' && size > 0) {
                poll(queue, --size);
            } else if (size > 0) {
                pollLast(queue, --size);
            }
        }
        if (size > 0) {
            return new int[] { peekLast(queue, size), peek(queue) };
        }
        return new int[] { 0, 0 };
    }

    int parseInt(String s) {
        int index;
        int length = s.length();
        int result = 0;

        if (s.charAt(2) == '-') {
            index = 3;
            for (;;) {
                result -= s.charAt(index++) - 48;
                if (index < length) {
                    result *= 10;
                } else {
                    return result;
                }
            }
        }
        index = 2;
        for (;;) {
            result += s.charAt(index++) - 48;
            if (index < length) {
                result *= 10;
            } else {
                return result;
            }
        }
    }

    void shiftUp(int[] queue, int start, int end) {
        while (end > start) {
            queue[end--] = queue[end];
        }
    }

    void shiftDown(int[] queue, int start, int end) {
        while (start < end) {
            queue[start++] = queue[start];
        }
    }

    void add(int[] queue, int size, int e) {
        if (size == 0) {
            queue[0] = e;
        } else {
            int index = findIndex(queue, e, 0, size);

            shiftUp(queue, index, size);
            queue[index] = e;
        }
    }

    int poll(int[] queue, int size) {
        int result = queue[0];

        shiftDown(queue, 0, size);
        return result;
    }

    int peek(int[] queue) {
        return queue[0];
    }

    int pollLast(int[] queue, int size) {
        return queue[size];
    }

    int peekLast(int[] queue, int size) {
        return queue[size - 1];
    }

    int findIndex(int[] queue, int e, int start, int end) {
        int mid;

        /* parametric search */
        while (start < end) {
            mid = (start + end - 1) / 2;
            if (queue[mid] <= e) {
                start = mid + 1;
            } else {
                end = mid;
            }
        }
        return start;
    }
}
