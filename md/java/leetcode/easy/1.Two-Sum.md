class Solution {
    public int[] twoSum(int[] nums, int target) {
        int length = nums.length;
        int prev =  nums[0];
        int curr;

        for (int i = 1; i < length; i++) {
            curr = nums[i];
            if (prev + curr == target) {
                return new int[] { i - 1, i };
            }
            prev = curr;
        }
        for (int i = 0;; i++) {
            prev = nums[i];
            for (int j = i + 2; j < length; j++) {
                if (prev + nums[j] == target) {
                    return new int[] { i, j };
                }
            }
        }
    }
}
