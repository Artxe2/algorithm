class Solution {
    public int solution(int[] nums) {
        int a = nums.length;
        int b = a - 1;
        int c = b - 1;
        int n;
        int answer = 0;

        for (int i = 0; i < c; i++) {
            for (int j = i + 1; j < b; j++) {
                for (int k = j + 1; k < a; k++) {
                    n = nums[i] + nums[j] + nums[k];
                    if (n % 2 > 0 && verifyPrimeNumber(n)) {
                        answer++;
                    }
                }
            }
        }
        return answer;
    }

    boolean verifyPrimeNumber(int n) {
        double sqrt = Math.sqrt(n);

        for (int i = 3; i <= sqrt; i += 2) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}
