class Solution {
    public int solution(int[][] maps) {
        int yLength = maps.length - 1;
        int xLength = maps[0].length - 1;
        int y;
        int x;
        int time;
        int[] cell;
        java.util.Deque<int[]> cells =
                new java.util.ArrayDeque<>(yLength * xLength);

        maps[0][0] = 0;
        cells.add(new int[] { 0, 0, 1 });
        do {
            cell = cells.pop();
            y = cell[0] + 1;
            x = cell[1];
            time = cell[2] + 1;
            if (y <= yLength && maps[y][x] == 1) {
                if (y == yLength && x == xLength) {
                    return time;
                }
                maps[y][x] = time;
                cells.add(new int[] { y, x, time });
            }
            y -= 2;
            if (y >= 0 && maps[y][x] == 1) {
                if (y == yLength && x == xLength) {
                    return time;
                }
                maps[y][x] = time;
                cells.add(new int[] { y, x, time });
            }
            y++;
            x++;
            if (x <= xLength && maps[y][x] == 1) {
                if (y == yLength && x == xLength) {
                    return time;
                }
                maps[y][x] = time;
                cells.add(new int[] { y, x, time });
            }
            x -= 2;
            if (x >= 0 && maps[y][x] == 1) {
                if (y == yLength && x == xLength) {
                    return time;
                }
                maps[y][x] = time;
                cells.add(new int[] { y, x, time });
            }
        } while (!cells.isEmpty());
        return -1;
    }
}
