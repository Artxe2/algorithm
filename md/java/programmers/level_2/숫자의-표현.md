class Solution {
    public int solution(int n) {
        if (n == 1) {
            return 1;
        }

        int length = n / 2;
        int answer = n % 2 == 0 ? 1 : 2;

        for (int i = 3; i <= length; i += 2) {
            if (n % i == 0) {
                answer++;
            }
        }
        return answer;
    }
}
