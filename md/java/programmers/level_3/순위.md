class Solution {
    public int solution(int n, int[][] results) {
        int answer = n;
        boolean[][] match = new boolean[n + 1][n + 1];

        for (int i = 1; i <= n; i++) {
            match[i][i] = true;
        }
        for (int[] a : results) {
            match[a[0]][a[1]] = true;
        }
        matchInference(n, match);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (!match[i][j] && !match[j][i]) {
                    answer--;
                    break;
                }
            }
        }
        return answer;
    }

    void matchInference(int n, boolean[][] match) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (match[j][i] && i != j) {
                    /* inference */
                    for (int k = 1; k <= n; k++) {
                        if (match[i][k]) {
                            match[j][k] = true;
                        }
                    }
                }
            }
        }
    }
}
