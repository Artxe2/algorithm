class Solution {
    public long solution(int n, int[] times) {
        long start = 1;
        long mid;
        long end = 0;
        long acc;

        for (int i : times) {
            end += i;
        }
        while (start < end) {
            acc = 0;
            mid = (start + end - 1) / 2;
            for (int i : times) {
                acc += mid / i;
            }
            if (acc >= n) {
                end = mid;
            } else {
                start = mid + 1;
            }
        }
        return end;
    }
}
